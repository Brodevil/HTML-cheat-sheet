handler.setFormatter(logging.Formatter('%(asctime)s:%(levelname)s:%(name)s: %(message)s'))
logger.addHandler(handler)


def search(obj, from_obj):
    for i in obj:
        for j in from_obj:
            if i.lower() == j.lower():
                return True
                break
    else:
        return False


client = discord.Client

        def is_correct(m):

        answer = random.randint(1, 10)

        try:
